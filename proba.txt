txtA = Element("txtA"); txtB = Element("txtB"); txtC = Element("txtC"); message = Element("msg")
result_x = Element("x"); result_x1 = Element("x1"); result_x2 = Element("x2")

def solve(*args, **kwargs):
    a = int(txtA.value)
    b = int(txtB.value)
    c = int(txtC.value)
    if a == 0:
        x = -c / b
        message.write(f"This equation is NOT quadratic!")
        result_x.write(x)
    elif b**2 - 4*a*c < 0:
        message.write("This equation has NO result!")
    else:
        x1 = (-b + (b**2 - 4*a*c)**0.5) / (2*a)
        x2 = (-b + (b**2 - 4*a*c)**0.5) / (2*a)
        result_x1.write(x1)
        result_x2.write(x2)
